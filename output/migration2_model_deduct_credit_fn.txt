DECLARE current_credits INTEGER;
BEGIN
  SELECT credits_left INTO current_credits
  FROM user_profiles WHERE id = p_user_id FOR UPDATE;
  IF current_credits IS NULL OR current_credits <= 0 THEN
    RETURN FALSE;
  END IF;
  UPDATE user_profiles
  SET credits_left = credits_left - 1
  WHERE id = p_user_id;
  INSERT INTO credit_ledger(user_id, delta, reason, balance)
  VALUES(p_user_id, -1, p_reason, current_credits - 1);
  RETURN TRUE;
END; $$;
