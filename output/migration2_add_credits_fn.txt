    ) RETURNS void LANGUAGE plpgsql AS $$
    DECLARE new_balance INTEGER;
    BEGIN

        SET credits_left = credits_left + p_amount
          WHERE id = p_user_id
            RETURNING credits_left INTO new_balance;
              INSERT INTO credit_ledger(user_id, delta, reason, balance)
                VALUES(p_user_id, p_amount, p_reason, new_balance);
                END; $$;
